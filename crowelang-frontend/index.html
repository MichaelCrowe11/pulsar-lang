pipeline DataProcessor<T> {
    @parallel(workers: 8)
    transform(data: T[]): ProcessedData[] {
        return data.map(item => {
            return this.processItem(item);
        });
    }
    
    @memoize
    processItem(item: T): ProcessedData {
        // Expensive computation with automatic caching
        return { 
            id: item.id,
            result: complexCalculation(item)
        };
    }
}

@gpu_accelerated
class NeuralNetwork {
    layers: Layer[];
    
    constructor(architecture: number[]) {
        this.layers = architecture.map((size, i) => 
            new DenseLayer(size, i > 0 ? architecture[i-1] : size)
        );
    }
    
    @tensor
    forward(input: Matrix): Matrix {
        return this.layers.reduce((data, layer) => 
            layer.forward(data), input
        );
    }
    
    @optimize("adam")
    train(dataset: TrainingData[], epochs: number): void {
        // Automatic differentiation and optimization
        for (let epoch = 0; epoch < epochs; epoch++) {
            this.trainEpoch(dataset);
        }
    }
}
